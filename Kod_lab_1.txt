void print(Data* str_data, int n)
{
	int fev, mm_test;

	int mm;

	cout << endl;
	for (int i = 0; i < n; i++)
	{
		if (str_data[i].yy % 4 == 0) //високосный год
			fev = 1;
		else fev = 0;


		cout << str_data[i].dd << ":"; //дата
		cout << str_data[i].mm << ":";
		cout << str_data[i].yy << "    ";

		if (str_data[i].mm == 2) //поределяем месяц 
			mm_test = 28 + fev;
		else if (str_data[i].mm == 4 || str_data[i].mm == 6 || str_data[i].mm == 9 || str_data[i].mm == 11)
			mm_test = 30;
		else mm_test = 31;

		if (str_data[i].dd < mm_test)                //следующая дата
		{
			cout << str_data[i].dd + 1 << ":";
			cout << str_data[i].mm << ":";
			cout << str_data[i].yy << "    ";
		}

		if (str_data[i].dd >= mm_test)
		{
			if (str_data[i].mm == 12)
			{
				cout << "1:";
				cout << "1:";
				cout << str_data[i].yy + 1 << "    ";
			}
			if (str_data[i].mm < 12)
			{
				cout << "1:";
				cout << str_data[i].mm + 1 << ":";
				cout << str_data[i].yy << "    ";
			}
		}

		if ((str_data[i].dd - 1) > 1) //предыдущая дата
		{
			cout << str_data[i].dd - 1 << ":";
			cout << str_data[i].mm << ":";
			cout << str_data[i].yy << endl;
		}

		if ((str_data[i].dd - 1) <= 1)
		{
			if (str_data[i].mm == 1)
			{
				cout << "31:";
				cout << "12:";
				cout << str_data[i].yy - 1 << endl;
			}
			if (str_data[i].mm > 1)
			{
				mm = str_data[i].mm - 1;

				if (mm == 2) //определяем месяц 
					mm_test = 28 + fev;
				else if (mm == 4 || mm == 6 || mm == 9 || mm == 11)
					mm_test = 30;
				else mm_test = 31;

				cout << mm_test << ":";
				cout << mm << ":";
				cout << str_data[i].yy << endl;
			}
		}
	}
}


int col_dates() //подсчитывает количество дат
{
	int m = 0, n = 0;
	ifstream File;
	File.open("1.txt");
	if (!File)
	{
		cout << "Ошибка!" << endl;
		return 0;
	}
	while (File >> n)
	{
		m++;
	}
	File.close();
	return m / 3;
}
